# 1. Crear la estructura de carpetas mkdir -p .github/workflows # 2. Crear package.json cat << 'EOF' > package.json { "name": "simulador-algebra", "version": "1.0.0", "description": "Simulador de Álgebra Lineal 100% Offline", "main": "index.js", "scripts": { "build": "mkdir -p www && cp index.html www/ && cp -r node_modules/mathjax/es5 www/mathjax" }, "dependencies": { "@capacitor/android": "^6.0.0", "@capacitor/cli": "^6.0.0", "@capacitor/core": "^6.0.0", "mathjax": "^3.2.2" } } EOF # 3. Crear capacitor.config.json cat << 'EOF' > capacitor.config.json { "appId": "com.simulador.algebralineal", "appName": "Simulador Algebra", "webDir": "www", "bundledWebRuntime": false } EOF # 4. Crear automatización para GitHub Actions cat << 'EOF' > .github/workflows/build-apk.yml name: Compilar APK de Android on: push: branches: - main workflow_dispatch: jobs: build: runs-on: ubuntu-latest steps: - name: Descargar Código uses: actions/checkout@v4 - name: Configurar Node.js uses: actions/setup-node@v4 with: node-version: 18 - name: Configurar Java JDK 17 uses: actions/setup-java@v4 with: distribution: 'zulu' java-version: '17' - name: Configurar Android SDK uses: android-actions/setup-android@v3 - name: Instalar Dependencias run: npm install - name: Generar Archivos Web run: npm run build - name: Agregar Plataforma Android Capacitor run: | npx cap add android npx cap copy android - name: Compilar APK con Gradle run: | cd android chmod +x gradlew ./gradlew assembleDebug - name: Subir APK como Artefacto Descargable uses: actions/upload-artifact@v4 with: name: Simulador-Algebra-Offline-APK path: android/app/build/outputs/apk/debug/app-debug.apk EOF # 5. Crear index.html con las 120 preguntas y MathJax Local cat << 'EOF' > index.html Simulador Completo de Álgebra Lineal - 120 Preguntas

Simulador Álgebra Lineal

120 Preguntas Interactivas - Modo Offline

0 Correctas
0 Incorrectas
0 / 120 Progreso
Bloque 1: Vectores Pregunta 1 / 120
Cargando pregunta...
    EOF echo "--------------------------------------------------------" echo "¡LISTO! Todos los archivos necesarios han sido creados." echo "Ahora sube la carpeta a GitHub para compilar tu APK." echo "--------------------------------------------------------"
